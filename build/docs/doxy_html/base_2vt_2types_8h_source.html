167 // The following preprocessor code produces typedefs for VtArray holding
168 // various scalar value types. The produced typedefs are of the form:
169 //
170 // typedef VtArray<int> VtIntArray;
171 // typedef VtArray<double> VtDoubleArray;
172 template<typename T> class VtArray;
173 #define VT_ARRAY_TYPEDEF(r, unused, elem) \
174 typedef VtArray< VT_TYPE(elem) > \
175 TF_PP_CAT(Vt, TF_PP_CAT(VT_TYPE_NAME(elem), Array)) ;

277 // Shape representation used in VtArray for legacy code. This is not supported
278 // at the pxr level or in usd. Shape is represented by a total size, plus sized
279 // dimensions other than the last. The size of the last dimension is computed
280 // as totalSize / (product-of-other-dimensions).
281 struct Vt_ShapeData {
282  unsigned int GetRank() const {
283  return
284  otherDims[0] == 0 ? 1 :
285  otherDims[1] == 0 ? 2 :
286  otherDims[2] == 0 ? 3 : 4;
287  }
288  bool operator==(Vt_ShapeData const &other) const {
289  if (totalSize != other.totalSize)
290  return false;
291  unsigned int thisRank = GetRank(), otherRank = other.GetRank();
292  if (thisRank != otherRank)
293  return false;
294  return std::equal(otherDims, otherDims + GetRank() - 1,
295  other.otherDims);
296  }
297  bool operator!=(Vt_ShapeData const &other) const {
298  return !(*this == other);
299  }
300  void clear() {
301  memset(this, 0, sizeof(*this));
302  }
303  static const int NumOtherDims = 3;
304  size_t totalSize;
305  unsigned int otherDims[NumOtherDims];
306 };
